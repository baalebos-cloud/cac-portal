@auth.route("/login", methods=["GET", "POST"]) def login(): if request.method == "POST": email = request.form["email"] password = request.form["password"] user = User.query.filter_by(email=email).first() if not user or not user.check_password(password): flash("Invalid credentials", "danger") return redirect(url_for("auth.login")) login_user(user) # Redirect by role if user.is_admin: return redirect(url_for("admin.dashboard")) else: return redirect(url_for("user.dashboard")) return render_template("login.html")
